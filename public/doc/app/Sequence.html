def self.find_col_for(model,col)
  ret=nil
  find_all.each do |line|
      tokens=line.utility.split(".")
      ##puts "Sequence.find_col_for:"+model+"."+col+":"+line.utility
      if(tokens[0]==model && tokens[1]==col)
        ret=line
      end
  end
  ret
end

def self.find_cols_for(model)
  ret=[]
  find_all.each do |line|
    tokens=line.split(".")[0]
    if(tokens[0]==model)
      ret<<tokens[1]
      end
  end
  ret
end
