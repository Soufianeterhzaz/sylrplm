def self.get_conditions(filter)
        filter = filters.gsub("*","%")
        ret={}
        unless filter.nil?
                ret[:qry] = "ident LIKE :v_filter or "+qry_type+" or designation LIKE :v_filter or "+qry_status+
                " or "+qry_owner+" or date LIKE :v_filter "
                ret[:values]={:v_filter => filter}
        end
        ret

end
